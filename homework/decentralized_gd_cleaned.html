# Calculate the objective function value
# Inputs:
#     w: d-by-1 matrix
#     x: n-by-d matrix
#     y: n-by-1 matrix
#     lam: scalar, the regularization parameter
# Return:
#     objective function value (scalar)
def objective(w, x, y, lam):
    n, d = x.shape
    yx = numpy.multiply(y, x) # n-by-d matrix
    yxw = numpy.dot(yx, w) # n-by-1 matrix
    vec1 = numpy.exp(-yxw) # n-by-1 matrix
    vec2 = numpy.log(1 + vec1) # n-by-1 matrix
    loss = numpy.mean(vec2) # scalar
    reg = lam / 2 * numpy.sum(w * w) # scalar
    return loss + reg

# create worker class 
# alpha = 1E-3
class Worker:
    def __init__(self, x, y): 
        self.x = x #s-by-d local feature matrix
        self.y = y #s-by-1 local label matrix
        self.s = x.shape[0] # number of local samples
        self.d = x.shape[1] # number of features
        self.w = numpy.zeros((d, 1)) #d-by-1 model parameter vector
        self.g = numpy.zeros((self.d,1)) # the gradient of the model
        self.loss = 0   # the loss function value of the model
        
        
#set the model parameters to the latest
    def set_param(self, w):
        self.w = w
            # compute the local loss
    def loss0(self):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(-yxw) # s-by-1 matrix
        vec2 = numpy.log(1 + vec1) # s-by-1 matrix
        self.loss = numpy.sum(vec2) # loss function
        return self.loss

    # compute the local gradient
    def gradient(self):
        yx = numpy.multiply(self.y, self.x) # s-by-d matrix
        yxw = numpy.dot(yx, self.w) # s-by-1 matrix
        vec1 = numpy.exp(yxw) # s-by-1 matrix
        vec2 = numpy.divide(yx, 1 + vec1) #s-by-d matrix
        self.g = -numpy.sum(vec2, axis = 0).reshape(self.d, 1) #d-by-1 matrix
        return self.g
    
    def gd(self, gradient):
        self.w -= alpha * gradient

# create worker
def call_worker(m, x, y):
    n, d = x.shape
    s = math.floor(n/m) # number of samples in each worker node
    
    workers = []
    for i in range(m):
        indices = list(range(i*s, (i+1)*s))
        worker = Worker(x[indices,:], y[indices,:])
        workers.append(worker)
        
    return workers

for t in range(max_epoch):
    
    gradients_aver = []
    losses_aver = []
    gradients_hi = []
    losses_hi = []
    for i in range(m):
        gradient_aver = workers_avg[i].gradient()
        gradients_aver.append(gradient_aver)
        loss_aver = workers_avg[i].loss0()
        losses_aver.append(loss_aver)
        
        gradient_hi = workers_high[i].gradient()
        gradients_hi.append(gradient_hi)
        loss_hi = workers_high[i].loss0()
        losses_hi.append(loss_hi)
    
    
    
    #########
    # average matrix:
    # put gradient and loss from each worker in a list
    gradient_aver = numpy.array(gradients_aver).reshape(m, d) # m-by-d matrix
    loss_aver = numpy.array(losses_aver).reshape(m, 1) # m-by-1 vector
    
    gradient_avg = numpy.dot(average_weighted_mat, gradient_aver)
    loss_avg = numpy.dot(average_weighted_mat, loss_aver)
    
    # calculate the average gradient and loss of each epoch
    gradient_avg = numpy.sum(gradient_avg, axis=0).reshape(d,1) # d-by-1 vector, the sum of gradient of all samples in each epoch
    loss_avg = numpy.sum(loss_avg, axis=0) # scalar, the loss of all samples in each epoch
    
    gradient_avg /= n
    loss_avg /= n

    loss_average_l.append(loss_avg)
    
    
    #########
    # weighted matrix:
    # put gradient and loss from each worker in a list
    gradient_hi = numpy.array(gradients_hi).reshape(m, d) # m-by-d matrix
    loss_hi = numpy.array(losses_hi).reshape(m, 1) # m-by-1 vector
      
    gradient_high = numpy.dot(high_weighted_mat, gradient_hi)
    loss_high = numpy.dot(high_weighted_mat, loss_hi)
    
    # calculate the average gradient and loss of each epoch

    loss_high = numpy.sum(loss_high, axis=0) # scalar, the loss of all samples in each epoch
    
    gradient_high /= n
    loss_high /= n

    loss_high_l.append(loss_high)
    
    
    print('Objective value of average weighted'+ str(loss_avg), ', Objective value of high weighted'+ str(loss_high))
    
    for i in range(m):
        workers_avg[i].gd(gradient_avg)
        workers_high[i].gd(gradient_high)
